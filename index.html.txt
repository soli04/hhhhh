    const r=await fetch(ep), d=await r.json();

    img.onload=()=>document.getElementById(cardId).classList.add('show');
    img.onerror=()=>document.getElementById(cardId).style.display='none';

  } catch { document.getElementById(cardId).style.display='none'; }
}

const sel={};
document.querySelectorAll('.chip').forEach(c=>{
  c.addEventListener('click',()=>{
    c.classList.toggle('sel');
    const g=c.dataset.g, t=c.querySelector('span').innerText.trim();
    if(!sel[g]) sel[g]=new Set();
    c.classList.contains('sel')?sel[g].add(t):sel[g].delete(t);
    updateCnt();
  });
});
function updateCnt(){
  const c1=['genre','mood'].reduce((s,g)=>s+(sel[g]?.size||0),0);
  const c2=['hero','el','watch'].reduce((s,g)=>s+(sel[g]?.size||0),0);
  document.getElementById('cnt1').textContent=c1;
  document.getElementById('cnt2').textContent=c2;
  document.getElementById('n1').disabled=c1===0;
  document.getElementById('n2').disabled=c2===0;
}

/* ── NAV ── */
let cur=1;
function goStep(n){
  document.getElementById(`step${cur}`)?.classList.remove('active');
  document.getElementById('loading').classList.remove('active');
  cur=n;
  document.getElementById(`step${n}`)?.classList.add('active');
  for(let i=1;i<=3;i++){const d=document.getElementById('d'+i);d.className='dot'+(i<n?' done':i===n?' active':'');}
  window.scrollTo({top:0,behavior:'smooth'});
}

/* ── RECOMMEND ── */
const LMSGS=['키워드 분석 중...','최적의 애니 탐색 중...','추천 목록 정리 중...','마지막 검토 중...'];
let lint;

function getKW(){
  const all=[];
  for(const s of Object.values(sel)) s.forEach(v=>all.push(v));
  return all;
}

async function recommend(){
  const kw=getKW(); if(!kw.length) return;
  document.getElementById('step2').classList.remove('active');
  document.getElementById('loading').classList.add('active');
  for(let i=1;i<=3;i++){const d=document.getElementById('d'+i);d.className='dot'+(i<3?' done':' active');}

  let mi=0;
  document.getElementById('lmsg').textContent=LMSGS[0];
  lint=setInterval(()=>{ mi=(mi+1)%LMSGS.length; document.getElementById('lmsg').textContent=LMSGS[mi]; },1600);

  const prompt=`너는 애니메이션 추천 전문가야. 사용자 취향 키워드: ${kw.join(', ')}

아래 JSON 형식으로만 답해. 다른 텍스트 절대 없이:
{"recommendations":[{"title":"한국어제목","titleJp":"일본어원제","year":"연도","episodes":"화수","genres":["장르"],"matchScore":90,"description":"2문장 소개","matchReason":"이 취향에 맞는 이유 한 줄"}]}

5편 추천. matchScore는 75~98 정수. 실제 존재하는 유명 애니만.`;

  try {
    const res=await fetch('https://api.anthropic.com/v1/messages',{
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body:JSON.stringify({
        model:'claude-sonnet-4-20250514',
        max_tokens:900,
        messages:[{role:'user',content:prompt}]
      })
    });
    clearInterval(lint);
    const data=await res.json();
    if(data.error) throw new Error(data.error.message||'API error');
    const raw=(data.content||[]).map(b=>b.text||'').join('');
    // Extract JSON even if there's extra text
    const match=raw.match(/\{[\s\S]*\}/);
    if(!match) throw new Error('JSON 파싱 실패');
    const parsed=JSON.parse(match[0]);
    document.getElementById('loading').classList.remove('active');
    showResults(parsed.recommendations,kw);
    document.getElementById('step3').classList.add('active');
    loadChars();
  } catch(e){
    clearInterval(lint);
    document.getElementById('loading').classList.remove('active');
    document.getElementById('step2').classList.add('active');
    for(let i=1;i<=3;i++){const d=document.getElementById('d'+i);d.className='dot'+(i<2?' done':i===2?' active':'');}
    let eb=document.getElementById('aerr');
    if(!eb){eb=document.createElement('div');eb.id='aerr';eb.className='err';document.querySelector('#step2 .glass').appendChild(eb);}
    eb.textContent='오류: '+e.message;
  }
}

function showResults(list,kw){
  document.getElementById('schips').innerHTML=kw.map(k=>`<div class="sum-chip">${k}</div>`).join('');
  document.getElementById('rgrid').innerHTML=list.map((a,i)=>`
    <div class="card">
      <div class="rank">0${i+1}</div>
      <div class="ct">${a.title||''}</div>
      ${a.titleJp?`<div class="cj">${a.titleJp}</div>`:''}
      <div class="tags">
        ${(a.genres||[]).map(g=>`<span class="tag g">${g}</span>`).join('')}
        ${a.year?`<span class="tag i">📅 ${a.year}</span>`:''}
        ${a.episodes?`<span class="tag i">🎬 ${a.episodes}화</span>`:''}
      </div>
      <div class="desc">${a.description||''}</div>
      ${a.matchReason?`<div class="reason">💡 ${a.matchReason}</div>`:''}
      <div class="bar-wrap">
        <span>일치도</span>
        <div class="mbar"><div class="mfill" data-p="${a.matchScore||80}"></div></div>
        <span class="mpct">${a.matchScore||80}%</span>
      </div>
    </div>
  `).join('');
  setTimeout(()=>document.querySelectorAll('.mfill').forEach(el=>el.style.width=el.dataset.p+'%'),80);
}

function reset(){
  Object.keys(sel).forEach(k=>sel[k]=new Set());
  document.querySelectorAll('.chip').forEach(c=>c.classList.remove('sel'));
  document.getElementById('step3').classList.remove('active');
  document.getElementById('aerr')?.remove();
  updateCnt();
  goStep(1);
  loadChars();
}
